cmake_minimum_required(VERSION 3.12)
project(yredis)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(YREDIS_READLINE_LIB "")

# 1) GNU readline (Linux, macOS Homebrew, MSYS2/Cygwin on Windows)
find_library(READLINE_LIBRARY readline)
if(READLINE_LIBRARY)
    find_path(READLINE_INCLUDE_DIR readline/readline.h)
    if(READLINE_INCLUDE_DIR)
        message(STATUS "yredis: found GNU readline at ${READLINE_LIBRARY}")
        set(YREDIS_READLINE_LIB ${READLINE_LIBRARY})
        set(YREDIS_READLINE_INC ${READLINE_INCLUDE_DIR})
        set(YREDIS_USE_READLINE 1)
    endif()
endif()

# 2) libedit (macOS system default, some Linux distros)
if(NOT YREDIS_USE_READLINE)
    find_library(EDITLINE_LIBRARY edit)
    if(EDITLINE_LIBRARY)
        find_path(EDITLINE_INCLUDE_DIR readline/readline.h HINTS /usr/include/editline /opt/homebrew/include/editline /usr/local/include/editline)
        if(EDITLINE_INCLUDE_DIR)
            message(STATUS "yredis: found libedit at ${EDITLINE_LIBRARY}")
            set(YREDIS_READLINE_LIB ${EDITLINE_LIBRARY})
            set(YREDIS_READLINE_INC ${EDITLINE_INCLUDE_DIR})
            set(YREDIS_USE_EDITLINE 1)
        endif()
    endif()
endif()

# 3) Windows native — no readline, pure getline fallback
if(NOT YREDIS_USE_READLINE AND NOT YREDIS_USE_EDITLINE)
    if(WIN32)
        message(STATUS "yredis: no readline/editline found, using getline fallback (Windows)")
    else()
        message(STATUS "yredis: no readline/editline found, using getline fallback")
    endif()
endif()

add_executable(yredis src/main.cpp src/repl.cpp src/formatter.cpp src/input.cpp)
target_link_libraries(yredis PUBLIC redisCli Core)

if(YREDIS_USE_READLINE)
    target_compile_definitions(yredis PUBLIC HAVE_READLINE)
    target_link_libraries(yredis PUBLIC ${YREDIS_READLINE_LIB})
    target_include_directories(yredis PUBLIC ${YREDIS_READLINE_INC})
elseif(YREDIS_USE_EDITLINE)
    target_compile_definitions(yredis PUBLIC HAVE_EDITLINE)
    target_link_libraries(yredis PUBLIC ${YREDIS_READLINE_LIB})
    target_include_directories(yredis PUBLIC ${YREDIS_READLINE_INC})
endif()

target_include_directories(yredis PUBLIC "${CMAKE_CURRENT_LIST_DIR}/src")
