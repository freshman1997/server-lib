if(TARGET PluginHost)
    add_executable(test_plugin_contracts test_plugin_contracts.cpp)
    add_executable(test_plugin_governance test_plugin_governance.cpp)
    target_link_libraries(test_plugin_contracts PUBLIC App PluginHost)
    target_link_libraries(test_plugin_governance PUBLIC App PluginHost)

    add_executable(test_plugin test_plugin.cpp)
    add_executable(test_lua_plugin test_lua_plugin.cpp)
    target_link_libraries(test_plugin PUBLIC App PluginHost PluginBridge ServerServices)
    target_link_libraries(test_lua_plugin PUBLIC App PluginHost PluginLua)
    if(TARGET HelloWorld)
        add_dependencies(test_plugin HelloWorld)
    endif()

    function(yuan_copy_plugin_runtime_dlls target_name)
        if(WIN32)
            foreach(dep IN ITEMS PluginCore PluginLua PluginTypeScript)
                if(TARGET ${dep})
                    add_custom_command(TARGET ${target_name} POST_BUILD
                        COMMAND ${CMAKE_COMMAND} -E copy_if_different
                                $<TARGET_FILE:${dep}>
                                $<TARGET_FILE_DIR:${target_name}>)
                endif()
            endforeach()
        endif()
    endfunction()

    yuan_copy_plugin_runtime_dlls(test_plugin_contracts)
    yuan_copy_plugin_runtime_dlls(test_plugin_governance)
    yuan_copy_plugin_runtime_dlls(test_plugin)
    yuan_copy_plugin_runtime_dlls(test_lua_plugin)

    if(BUILD_TESTING)
        add_test(NAME plugin_contracts COMMAND test_plugin_contracts)
        add_test(NAME plugin_governance COMMAND test_plugin_governance)
        if(TARGET HelloWorld)
            add_test(NAME plugin_host_smoke COMMAND test_plugin $<TARGET_FILE_DIR:HelloWorld>)
        else()
            add_test(NAME plugin_host_smoke COMMAND test_plugin)
        endif()
        add_test(NAME lua_plugin COMMAND test_lua_plugin)
    endif()
endif()
